<template>
    <button
      class="z-btn"
      :class="btnClass"
      @click="handleClick">

      <svg class="icon" aria-hidden="true" v-if="icon && !iconback">
        <use :xlink:href='newIcon'></use>
      </svg>

      <slot></slot>

      <svg class="icon iconback" aria-hidden="true" v-if="icon && iconback">
        <use :xlink:href='newIcon'></use>
      </svg>

      <svg class="icon iconback refresh" aria-hidden="true" v-if="loading">
        <use xlink:href='#z-refresh'></use>
      </svg>

    </button>

<script>
export default {
  name: 'z-button',
  props: {
    inline: {
      type: Boolean,
      default: false
    },
    disabled: {
      type: Boolean,
      default: false
    },
    plain: {
      type: Boolean,
      default: false
    },
    icon: {
      type: String,
      default: ''
    },
    iconback: {
      type: Boolean,
      default: false
    },
    loading: {
      type: Boolean,
      default: false
    }
  },
  computed: {
    btnClass () {
      return {
        'z-btn_active': this.active,
        'z-btn_plain': this.plain,
        'z-btn_inline': this.inline,
        'z-btn_disabled': this.disabled,
        'z-btn_loading': this.loading
      }
    },
    newIcon () {
      return `#${this.icon}`
    }
  },
  methods: {
    handleClick (event) {
      if (this.disabled) {
        event.preventDefault()
        event.stopPropagation()
        return false
      }
      this.$emit('click', event)
    }
  }
}
</script>

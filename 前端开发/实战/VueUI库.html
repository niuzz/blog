# vue.config.js

vue-cli 3.x之后，不再暴露webpack的配置文件，需要使用vue.config.js来合并webpack的配置

  • 入口 因为更改了默认目录结构，所以更改入口如下
pages: {
  index: {
    entry: 'site/main.js',
    template: 'public/index.html',
    filename: 'index.html'
  }
}

另外需要babel组件库，所以需要将 libs 目录添加到babel解析里

module.exports = {
  chainWebpack: config => {
    config.module
    .rule('js | jsx')
    .include
    .add('/libs')
    .end()
    .use('babel')
    .loader('babel-loader')
  }
}

# 编写组件

libs/components/button

<template>
    <button
      class="z-btn"
      :class="btnClass"
      @click="handleClick">

      <svg class="icon" aria-hidden="true" v-if="icon && !iconback">
        <use :xlink:href='newIcon'></use>
      </svg>

      <slot></slot>

      <svg class="icon iconback" aria-hidden="true" v-if="icon && iconback">
        <use :xlink:href='newIcon'></use>
      </svg>

      <svg class="icon iconback refresh" aria-hidden="true" v-if="loading">
        <use xlink:href='#z-refresh'></use>
      </svg>

    </button>
</template>

<script>
export default {
  name: 'z-button',
  props: {
    inline: {
      type: Boolean,
      default: false
    },
    disabled: {
      type: Boolean,
      default: false
    },
    plain: {
      type: Boolean,
      default: false
    },
    icon: {
      type: String,
      default: ''
    },
    iconback: {
      type: Boolean,
      default: false
    },
    loading: {
      type: Boolean,
      default: false
    }
  },
  computed: {
    btnClass () {
      return {
        'z-btn_active': this.active,
        'z-btn_plain': this.plain,
        'z-btn_inline': this.inline,
        'z-btn_disabled': this.disabled,
        'z-btn_loading': this.loading
      }
    },
    newIcon () {
      return `#${this.icon}`
    }
  },
  methods: {
    handleClick (event) {
      if (this.disabled) {
        event.preventDefault()
        event.stopPropagation()
        return false
      }
      this.$emit('click', event)
    }
  }
}
</script>

# 解析markdown

这个项目比较困难的一个点是解析markdown文件，既要将markdown文件生成vue，同时也要将markdown里面的相应演示代码，做渲染，进行展示。

经过多次尝试使用了vue-markdown-loader作为解析器。 因为需要解析展示的文档代码，还需要 markdown-it-container。

vue-markdown-loader负责将md文件转化成vue，而markdown-it-container用来对获取到的md文件内容进行操作。编写专门的展示代码的组件，用markdown-it-container替换成这个组件，渲染成vue的时候，就能够作为一个正常的组件从md --> vue进行展示

将组件各种属性进行分别渲染，不直观，所以写了两套方案，一套是分开渲染，一套是只高亮代码的。多做了保留

vue.config.js

function wrapCustomClass (render) {
  return function (...args) {
    return render(...args)
      .replace('<code class="', '<code class="hljs ')
      .replace('<code>', '<code class="hljs">')
  }
}
module.exports = {
  chainWebpack: config => {
    config.module
      .rule('js|jsx')
      .include
      .add('/libs')
      .end()
      .use('babel')
      .loader('babel-loader')
    config.module
      .rule('md')
      .test(/\.md/)
      .use('vue-loader')
      .loader('vue-loader')
      .end()
      .use('vue-markdown-loader')
      .loader('vue-markdown-loader/lib/markdown-compiler')
      .options({
        raw: true,
        preprocess: (MarkdownIt, source) => {
          MarkdownIt.renderer.rules.fence = wrapCustomClass(MarkdownIt.renderer.rules.fence) // 将md中的代码块高亮
          return source
        },
        use: [
          [require('markdown-it-container'), 'demo', { // 遇到demo的情况，处理成演示代码 和 代码高亮
            validate: (params) => params.trim().match(/^demo\s*(.*)$/),
            render: (tokens, idx) => {
              if (tokens[idx].nesting === 1) {
                // 1.获取第一行的内容使用markdown渲染html作为组件的描述
                let demoInfo = tokens[idx].info.trim().match(/^demo\s+(.*)$/)
                let description = (demoInfo && demoInfo.length > 1) ? demoInfo[1] : ''
                let descriptionHTML = description ? require('markdown-it')().render(description) : ''
                // 2.获取代码块内的html和js代码
                let content = tokens[idx + 1].content
                // 3.使用自定义开发组件【DemoBlock】来包裹内容并且渲染成案例和代码示例
                return `<demo-block>
                <div class="source" slot="source">${content}</div>
                ${descriptionHTML}
                <div class="hljs" slot="highlight">`
              } else {
                return '</div></demo-block>\n'
              }
            }
          }],
          [require('markdown-it-container'), 'code', {  // code情况，只处理代码高亮
            validate: (params) => params.trim().match(/^code\s*(.*)$/),
            render: (tokens, idx) => {
              if (tokens[idx].nesting === 1) {
                // 1.获取第一行的内容使用markdown渲染html作为组件的描述
                let demoInfo = tokens[idx].info.trim().match(/^code\s+(.*)$/)
                let description = (demoInfo && demoInfo.length > 1) ? demoInfo[1] : ''
                let descriptionHTML = description ? require('markdown-it')().render(description) : ''
                // 3.使用自定义开发组件【DemoBlock】来包裹内容并且渲染成案例和代码示例
                return `<demo-block>
                ${descriptionHTML}
                <div class="hljs" slot="highlight">`
              } else {
                return '</div></demo-block>\n'
              }
            }
          }]
        ]
      })
    config.resolve.alias
      .set('@', resolve('site'))
      .set('&', resolve('libs'))
    config.resolve.extensions
      .values(['.js', '.vue', '.json', '.css', '.md'])
  }
}

# 发布npm

经过上面的准备工作，我们已经成功编写了一个组件库，目前现在有一个button组件，我们将其打包成库文件

npm run lib

执行完之后，会将库打包到package目录下

npm login # 登录npmjs
npm publish # 发布包

现在可以看见库已经发布到npm上了，以后想要在项目中使用只需要 npm install即可

    # 绑定 this

    react 组件 this 是 undefined, 因此调用函数需要绑定，推荐在 constructor 中

    import React, { Component, Fragment } from "react";

class TodoList extends Component {
  constructor(props) {
    super(props);
    this.state = {
      inputValue: "",
      list: []
    };
    this.handleInputChange = this.handleInputChange.bind(this);
  }

  handleInputChange(e) {
    this.setState({
      inputValue: e.target.value
    });
  }

  handleBtnClick(e) {
    this.setState({
      list: [...this.state.list, this.state.inputValue],
      inputValue: ""
    });
  }

  handleLiClidk(index) {
    let list = [...this.state.list];
    list.splice(index, 1);
    this.setState({
      list
    });
  }

  render() {
    return (
      <Fragment>
        <input
          value={this.state.inputValue}
          onChange={this.handleInputChange}
        />
        <button onClick={this.handleBtnClick}>submit</button>
        <ul>
          {this.state.list.map((i, index) => {
            return (
              <li onClick={this.handleLiClidk.bind(this, index)} k={index}>
                {i}
              </li>
            );
          })}
        </ul>
      </Fragment>
    );
  }
}

export default TodoList;

    import React, { Component, Fragment } from "react";
import TodoItem from "./TodoItem";

class TodoList extends Component {
  constructor(props) {
    super(props);
    this.state = {
      inputValue: "",
      list: []
    };
    this.handleInputChange = this.handleInputChange.bind(this);
    this.handleBtnClick = this.handleBtnClick.bind(this);
    this.handleLiClidk = this.handleLiClidk.bind(this);
  }

  render() {
    return (
      <Fragment>
        <input
          value={this.state.inputValue}
          onChange={this.handleInputChange}
        />
        <button onClick={this.handleBtnClick}>submit</button>
        <ul>{this.getTodoItem()}</ul>
      </Fragment>
    );
  }

  handleBtnClick(e) {
    this.setState({
      list: [...this.state.list, this.state.inputValue],
      inputValue: ""
    });
  }

  handleLiClidk(index) {
    let list = [...this.state.list];
    list.splice(index, 1);
    this.setState({
      list
    });
  }

  handleInputChange(e) {
    this.setState({
      inputValue: e.target.value
    });
  }

  getTodoItem() {
    return this.state.list.map((i, index) => {
      return (
        <li>
          <TodoItem content={i} index={index} deleteItem={this.handleLiClidk} />
        </li>
      );
    });
  }
}

export default TodoList;

    import React, { Component } from "react";

class TodoItem extends Component {
  constructor(props) {
    super(props);
    this.handleClick = this.handleClick.bind(this);
  }
  render() {
    const { content, index } = this.props;
    return (
      <div onClick={this.handleClick} key={index}>
        {content}
      </div>
    );
  }
  handleClick() {
    const { deleteItem, index } = this.props;
    deleteItem(index);
  }
}

export default TodoItem;

    # react-transition-group

    import { CSSTransition } from "react-transition-group";


render() {
    return (
      <Fragment>
        <input
          value={this.state.inputValue}
          onChange={this.handleInputChange}
        />
        <button onClick={this.handleBtnClick}>submit</button>
        <ul>{this.getTodoItem()}</ul>
        <CSSTransition
          in={ this.state.show } // 控制开关
          timeout={ 300 } // 持续
          classNames='fade' // 类前缀
          unmountOnExit // 退出时是否移除dom
          onEntered={(el) => { el.style.color = '#f00'}} // 钩子函数
          appear = { true } // 第一次出现就带动画
        >
          <button className="animation-btn">animation</button>
        </CSSTransition>
        <button
          onClick={this.handleShow}
        >
          切换动画
        </button>
      </Fragment>
    );
  }

    .fade-enter, .fade-appear { /* fade-appear 为了第一次进来有动画*/
  opacity: 0;
}
.fade-enter-active, .fade-appear-active { /* fade-appear-active 为了第一次进来有动画*/
  opacity: 1;
  transition: all 1s;
}
.fade-enter-done {
  opacity: 1;
}
.fade-exit {

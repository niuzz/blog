中间件thunk，store.dispatch本身是接受一个对象 action, 如果使用了thunk，则可以发送一个函数当做action。

中间件是对dispatch方法的升级，中间指的是action到store之间. 判读参数，如果是对象就传给store，如果是函数，则执行函数。

saga将异步拆分到单独的文件。

就可以把componentDidMount中的复杂逻辑放到actionCreator中去管理

使用 redux-thunk 中间件, createStore 需要改造

store/index.js

import { createStore, applyMiddleware, compose } from "redux";
import reducer from "./reducer";
import thunk from 'redux-thunk';

const composeEnhancers =
  typeof window === "object" && window.__REDUX_DEVTOOLS_EXTENSION_COMPOSE__
    ? window.__REDUX_DEVTOOLS_EXTENSION_COMPOSE__({
        // Specify extension’s options like name, actionsBlacklist, actionsCreators, serialize...
      })
    : compose;

const enhancer = composeEnhancers(
  applyMiddleware(thunk)
);

const store = createStore(reducer, enhancer);

export default store;

actionCreator.js

import { INPUT_VALUE_CHANGE, ADD_ITEM, REMOVE_ITEM, INIT_LIST_ACTION } from './actionTypes'
import axios from 'axios'

export const getInputValueChangeAction = (value) => ({
  type: INPUT_VALUE_CHANGE,
  value
})

export const getSubmitAction = () => ({
  type: ADD_ITEM,
})

export const getItemClickAction = (index) => ({
  type: REMOVE_ITEM,
  value: index
})

export const initListAction = (data) => ({
  type: INIT_LIST_ACTION,
  data
})

export const getTodoList = () => {
  return (dispatch) => {
    axios.get('https://easy-mock.com/mock/596036509adc231f357bbada/example/list').then((res) => {
      const data = res.data.data;
      const action = initListAction(data);
      dispatch(action)
    })
  }
}

reducer.js

import { INPUT_VALUE_CHANGE, ADD_ITEM, REMOVE_ITEM, INIT_LIST_ACTION } from "./actionTypes";

const defaultState = {
  inputValue: "",
  list: []
};

const reducer = (state = defaultState, action) => {
  if (action.type === INPUT_VALUE_CHANGE) {
    const newState = JSON.parse(JSON.stringify(state));
    newState.inputValue = action.value;
    return newState;
  }
  if (action.type === ADD_ITEM) {
    const newState = JSON.parse(JSON.stringify(state));
    newState.list.push(newState.inputValue);
    newState.inputValue = '';    
    return newState;
  }
  if (action.type === REMOVE_ITEM) {
    const newState = JSON.parse(JSON.stringify(state));
    newState.list.splice(action.value, 1);
    return newState;
  }
  if (action.type === INIT_LIST_ACTION) {
    const newState = JSON.parse(JSON.stringify(state));
    newState.list = action.data;
    return newState;
  }
  return state;
};

export default reducer;

TodoList.js

import React, { Component } from "react";
import store from "./store";
import { getInputValueChangeAction, getSubmitAction, getItemClickAction, getTodoList } from "./store/actionCreator";
import TodoUI from "./TodoUI";

class TodoList extends Component {
  constructor(props) {
    super(props);

    this.handleInputValueChange = this.handleInputValueChange.bind(this);
    this.handleStoreChange = this.handleStoreChange.bind(this);
    this.handleSubmitClick = this.handleSubmitClick.bind(this);
    this.handleItemClick = this.handleItemClick.bind(this);
    this.state = store.getState();
    store.subscribe(this.handleStoreChange);
  }
  render() {
    return (
      <TodoUI
        inputValue={this.state.inputValue}
        inputValueChange={this.handleInputValueChange}
        submitClick={this.handleSubmitClick}
        itemClick={this.handleItemClick}
        list={this.state.list}
      ></TodoUI>
    );
  }
  componentDidMount() {
    const action = getTodoList()
    store.dispatch(action)
  }
  handleInputValueChange(e) {
    const value = e.target.value;
    const action = getInputValueChangeAction(value);
    store.dispatch(action);
  }
  handleSubmitClick() {
    const action = getSubmitAction();    
    store.dispatch(action);
  }
  handleStoreChange() {
    this.setState(store.getState());
  }
  handleItemClick(index) {
    const action = getItemClickAction(index);
    store.dispatch(action);
  }
}

export default TodoList;
